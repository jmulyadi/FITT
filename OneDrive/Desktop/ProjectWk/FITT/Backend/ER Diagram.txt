USERS ||--o{ WORKOUTS : performs
    USERS ||--o{ MEALS    : eats

    WORKOUTS ||--o{ EXERCISES : includes
    EXERCISES ||--o{ SETS      : records

    MEALS ||--o{ FOODS : consists_of

    USERS {
        uuid id PK
        string username UK
        int age
        string gender
        float weight
        float height
        string experience_level
        float bmi
    }

    WORKOUTS {
        int workout_id PK
        string username FK      %% Link back to USERS
        date date
        int duration            %% minutes
        int calories_burned
        string type             %% 'cardio' or 'strength'
        string cardio_type      %% only set if type = 'cardio'
        float distance          %% km -- only set if type = 'cardio'
    }

    EXERCISES {
        int exercise_id PK
        int workout_id FK       %% Link back to WORKOUTS
        string name
        string muscle_group
    }

    SETS {
        int set_id PK
        int exercise_id FK      %% Link back to EXERCISES
        int set_num
        int reps
        float weight            %% kg
        int intensity           %% 1=very light, 10=max effort
    }

    MEALS {
        int meal_id PK
        string username FK      %% Link back to USERS
        date date
        int meal_num            %% 1=Breakfast, 2=Lunch, 3=Dinner, etc.
        int calories_in
    }

    FOODS {
        int food_id PK
        int meal_id FK          %% Link back to MEALS
        string name
        string type
        int calories
    }
